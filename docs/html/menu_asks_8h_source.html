257template<class Sortable>
258int binarySearch(vector<Sortable> v, Sortable val) {
259 int l = 0, r = v.size() - 1;
260 while (l <= r) {
261 int m = l + (r - l) / 2;
262 if (v[m] == val) return m;
263 if (v[m] < val) l = m + 1;
264 else r = m - 1;
265 }
266 return -1;
267}
